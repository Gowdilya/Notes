function printL(n:number){
    for( var i = 0; i < n; i ++){
        console.log('L'+ '/n');
    }
     var lastLine = Array(i).fill('L');
     lastLine.push('/n');
     var lastLineStr = lastLine.join();
     console.log(lastLineStr);

}
printL(7);
