## set Dynamic Property via [ ]

```js
const property = "firsName";
const name = "Piyush Agarwal";

const user = {
  [property]: name,
};

console.log(user.firsName);
```


## Loop through properties
```js
const property = "firsName";
const name = "Piyush Agarwal";

const user = {
  [property]: name,
};

for (key in user){
    console.log(key);
    console.log(user[key])
}
```
